Add the following block, preceded by the comment `// === Phase 3: Settings Panel Toggle ===`:

```js
// === Phase 3: Settings Panel Toggle ===

const btnHamburger = document.getElementById('btn-hamburger')
const btnSettingsClose = document.getElementById('btn-settings-close')
const appEl = document.getElementById('app')

let settingsOpen = false

function openSettings() {
  settingsOpen = true
  appEl.classList.add('settings-open')
  btnHamburger.setAttribute('aria-label', 'Close settings')
  window.glorb.resize(440, 468)
}

function closeSettings() {
  settingsOpen = false
  appEl.classList.remove('settings-open')
  btnHamburger.setAttribute('aria-label', 'Open settings')
  window.glorb.resize(286, 468)
}

btnHamburger.addEventListener('click', () => {
  if (settingsOpen) {
    closeSettings()
  } else {
    openSettings()
  }
})

btnSettingsClose.addEventListener('click', closeSettings)
```

No changes to any existing JS. The Phase 2 timer engine block is untouched. The existing close/overlay handlers are untouched.

2. Add the following JS at the END of the `<script>` block in renderer.html (after the theme restore block), inside the existing `<script>` tag:
```js
// === Phase 5: Serial Status Dot ===

const serialDot = document.getElementById('serial-status-dot')
const serialLabel = document.getElementById('serial-status-label')

function applySerialStatus(connected) {
  if (connected) {
    serialDot.className = 'status-dot connected'
    serialLabel.textContent = 'Connected'
  } else {
    serialDot.className = 'status-dot disconnected'
    serialLabel.textContent = 'No device'
  }
}

// Query current state when settings panel opens
const _origOpenSettings = openSettings
function openSettings() {
  _origOpenSettings()
  window.glorb.serialStatus().then(({ connected }) => applySerialStatus(connected))
}

// React to connection changes while panel is open (D-18: push from main)
window.glorb.onSerialStatus(({ connected }) => {
  if (settingsOpen) applySerialStatus(connected)
})
```

Note: `openSettings` is already declared as a `function` (not const/let) earlier in the script, so wrapping it is safe. The override reassigns the name in the same scope.

After rotation, effective drawable canvas for each display: 64px wide × 128px tall (portrait).

U8g2 draw API used in this plan:
```cpp
// Oval (filled ellipse)
u8g2.drawEllipse(cx, cy, rx, ry, U8G2_DRAW_ALL);  // then fillEllipse or use filled variant

// For filled ellipse use:
u8g2.setDrawColor(1);
// U8g2 drawEllipse draws outline only — use drawFilledEllipse or loop drawLine approach
// Correct method for filled: u8g2.drawEllipse does outline; for filled use the approach below:
// u8g2_font_logisoso32_tr is available for SMILE character

// Standard sequence per display:

// ... draw calls ...
u8g2_hw.sendBuffer();
```

Note on filled ellipse: U8g2's `drawEllipse(cx, cy, rx, ry, U8G2_DRAW_ALL)` draws an outline.
For a filled oval, use the built-in filled variant: check U8g2 API — the method is
`u8g2.drawFilledEllipse(cx, cy, rx, ry, U8G2_DRAW_ALL)` if available in the version in use
(U8g2 2.35.7). If unavailable, fill by drawing stacked horizontal lines covering the ellipse area.
Verify in U8g2 source or use the horizontal-lines fallback to be safe.

**firmware/src/display.cpp:**
```cpp
#include <U8g2lib.h>
#include "display.h"

// Display 1: Hardware I2C on A4 (SDA) / A5 (SCL)
// Physical mount is CCW 90° rotated — U8G2_R1 corrects to portrait (64w x 128h canvas)
U8G2_SSD1306_128X64_NONAME_F_HW_I2C u8g2_hw(U8G2_R1, U8X8_PIN_NONE);

// Display 2: Software I2C on A2 (SDA=16) / A3 (SCL=17)
// Physical mount is CW 90° rotated — U8G2_R3 corrects to portrait (64w x 128h canvas)
U8G2_SSD1306_128X64_NONAME_F_SW_I2C u8g2_sw(U8G2_R3, /*clock=*/17, /*data=*/16, U8X8_PIN_NONE);

// Draw OPEN_EYES state to one U8g2 instance.
// Portrait canvas: 64px wide x 128px tall after rotation correction.
// Filled ellipse centered at (32, 64), rx=20, ry=30.
// U8g2 2.35.7 provides drawFilledEllipse for a filled oval.
// If drawFilledEllipse is not available, fall back to stacked horizontal lines:
//   for (int dy = -30; dy <= 30; dy++) {
//     int dx = (int)(20.0 * sqrt(1.0 - (dy * dy) / (30.0 * 30.0)));
//     u8g2.drawHLine(32 - dx, 64 + dy, 2 * dx);
//   }
static void drawEye(U8G2 &u8g2) {
    u8g2.setDrawColor(1);
    // Try filled ellipse — centered on 64x128 portrait canvas
    // cx=32, cy=64, rx=20 (horizontal radius), ry=30 (vertical radius)
    u8g2.drawFilledEllipse(32, 64, 20, 30, U8G2_DRAW_ALL);
}

// Draw SMILE state to one U8g2 instance.
// "^" character with logisoso32 font, centered on 64x128 portrait canvas.
// logisoso32 glyphs are ~32px tall. Center x approx: canvas_w/2 - char_w/2.
// Use getStrWidth to compute exact x offset after setting font.
static void drawSmile(U8G2 &u8g2) {
    u8g2.setFont(u8g2_font_logisoso32_tr);
    // "^" rendered: character width ~24px, height ~32px
    // Centered: x = (64 - charW) / 2, y = 64 + 16 (baseline below center)
    int charW = u8g2.getStrWidth("^");
    int x = (64 - charW) / 2;
    int y = 80;  // baseline position on 128px tall canvas — tune if needed
    u8g2.drawStr(x, y, "^");
}

void displaySetup() {
    u8g2_hw.begin();
    u8g2_sw.begin();
}

void showDisplay(DisplayState state) {
    // Update both displays in one call
    switch (state) {
        case DisplayState::OPEN_EYES:
            u8g2_hw.clearBuffer();
            drawEye(u8g2_hw);
            u8g2_hw.sendBuffer();

            u8g2_sw.clearBuffer();
            drawEye(u8g2_sw);
            u8g2_sw.sendBuffer();
            break;

        case DisplayState::SMILE:
            u8g2_hw.clearBuffer();
            drawSmile(u8g2_hw);
            u8g2_hw.sendBuffer();

            u8g2_sw.clearBuffer();
            drawSmile(u8g2_sw);
            u8g2_sw.sendBuffer();
            break;
    }
}
```

#include <Arduino.h>

void setup() {

|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Denial of Service | SW I2C bus (A2/A3) | accept | SW I2C is single-master, no bus contention possible; if display fails to init, displaySetup() silently continues — acceptable for embedded hardware |
| T-04-02 | Denial of Service | HW I2C bus (A4/A5) if two devices share address | accept | Both displays are on separate buses (HW vs SW I2C), so address collision is impossible by architecture |
| T-04-03 | Elevation of Privilege | N/A — no user input, no network | accept | Pure output-only firmware; no attack surface in this phase |

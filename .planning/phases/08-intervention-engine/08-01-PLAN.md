function clearAllTimers () {
  while (escalationTimers.length) {
    const ref = escalationTimers.pop()
    if (ref && ref._isInterval) clearInterval(ref)
    else clearTimeout(ref)
  }
  if (overlayWin && !overlayWin.isDestroyed()) {
    overlayWin.close()
    overlayWin = null
  }
}
